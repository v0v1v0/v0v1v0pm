d2<- (-2*logLik(modIDS)) + (2*logLik(modF))

vcs<- VarCorr(modF)

vcs$Series[2,2] + (2*vcs$Series[2,1])) + (vcs$Ind[1,1] + vcs$Ind[2,2] + (2*vcs$Ind[2,1])))
